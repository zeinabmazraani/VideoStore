pipeline {
    agent any

    triggers {
        // Check GitHub for new commits every 2 minutes
        pollSCM('H/2 * * * *')
    }

    environment {
        IMAGE_NAME = "videostore"
    }

    stages {

        stage('Checkout') {
            steps {
                echo '=== Checking out latest code from GitHub ==='
                cleanWs()
                git branch: 'main', url: 'https://github.com/zeinabmazraani/VideoStore'
            }
        }

        stage('Build in Minikube Docker') {
            steps {
                script {
                    env.IMAGE_TAG = "${BUILD_NUMBER}"  // unique tag per build
                    echo "=== Building Docker image: ${IMAGE_NAME}:${IMAGE_TAG} ==="

                    bat """
                    @echo off
                    call minikube docker-env --shell=cmd > docker_env.bat
                    call docker_env.bat
                    echo Building Docker image: ${IMAGE_NAME}:${IMAGE_TAG}
                    docker build -t ${IMAGE_NAME}:${IMAGE_TAG} .
                    """
                }
            }
        }

        stage('Deploy to Minikube') {
            steps {
                echo "=== Deploying image ${IMAGE_NAME}:${IMAGE_TAG} to Kubernetes ==="

                bat """
                @echo off
                call minikube docker-env --shell=cmd > docker_env.bat
                call docker_env.bat

                REM Force deployment to use the newly built image
                kubectl set image deployment/django-deployment django-container=${IMAGE_NAME}:${IMAGE_TAG} --record || (
                    echo Deployment not found, creating...
                    kubectl apply -f deployment.yaml
                )

                kubectl apply -f service.yaml

                REM Wait for rollout completion
                kubectl rollout restart deployment/django-deployment
                kubectl rollout status deployment/django-deployment
                """
            }
        }
    }

    post {
        success {
            echo " Deployment succeeded — running ${IMAGE_NAME}:${IMAGE_TAG}"
        }
        failure {
            echo " Deployment failed — check console output."
        }
    }
}
