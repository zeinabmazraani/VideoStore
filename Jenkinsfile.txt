pipeline {
    agent any

    // Automatically check GitHub every 2 minutes for new commits
    triggers {
        pollSCM('H/2 * * * *')
    }

    environment {
        // Jenkins build number is used to tag Docker images uniquely
        IMAGE_NAME = "videostore"
        IMAGE_TAG = "${BUILD_NUMBER}"
    }

    stages {

        stage('Checkout') {
            steps {
                echo '===  Checking out latest code from GitHub ==='
                cleanWs()
                git branch: 'main', url: 'https://github.com/zeinabmazraani/VideoStore'
            }
        }

        stage('Build in Minikube Docker') {
            steps {
                echo '===  Building Docker image inside Minikube ==='
                bat '''
                @echo off
                REM === Connect Docker to Minikube ===
                call minikube docker-env --shell=cmd > docker_env.bat
                call docker_env.bat

                REM === Wait for environment to load ===
                timeout /t 3 /nobreak >nul

                REM === Build Docker image with unique tag ===
                echo Building image: %IMAGE_NAME%:%IMAGE_TAG%
                docker build -t %IMAGE_NAME%:%IMAGE_TAG% . || exit /b 1
                '''
            }
        }

        stage('Deploy to Minikube') {
            steps {
                echo '===  Deploying new version to Kubernetes ==='
                bat '''
                @echo off

                REM === Use new image tag in deployment ===
                kubectl set image deployment/django-deployment django-container=%IMAGE_NAME%:%IMAGE_TAG% --record || exit /b 1

                REM === Wait for rollout to complete ===
                kubectl rollout status deployment/django-deployment || exit /b 1
                '''
            }
        }
    }

    post {
        success {
            echo ' Deployment completed successfully! Your web app should now show the latest changes.'
        }
        failure {
            echo ' Deployment failed â€” please check Jenkins console output and kubectl logs.'
        }
    }
}
