pipeline {
    agent any

    triggers {
        // Poll GitHub every 2 minutes
        pollSCM('H/2 * * * *')
    }

    environment {
        IMAGE_NAME = "videostore"
    }

    stages {

        stage('Checkout') {
            steps {
                echo '=== Checking out latest code from GitHub ==='
                cleanWs()
                git branch: 'main', url: 'https://github.com/zeinabmazraani/VideoStore'
            }
        }

        stage('Build in Minikube Docker') {
            steps {
                script {
                    env.IMAGE_TAG = "${BUILD_NUMBER}" // unique tag per build

                    echo "=== Building Docker image: ${IMAGE_NAME}:${IMAGE_TAG} ==="

                    bat """
                    @echo off
                    call minikube docker-env --shell=cmd > docker_env.bat
                    call docker_env.bat

                    REM === Clean any cached image to avoid reusing old layers ===
                    docker image rm ${IMAGE_NAME}:latest -f >nul 2>&1
                    docker image rm ${IMAGE_NAME}:${IMAGE_TAG} -f >nul 2>&1

                    REM === Build and tag new image ===
                    docker build -t ${IMAGE_NAME}:${IMAGE_TAG} -t ${IMAGE_NAME}:latest .
                    """
                }
            }
        }

        stage('Deploy to Minikube') {
            steps {
                echo "=== Deploying image ${IMAGE_NAME}:${IMAGE_TAG} ==="

                bat """
                @echo off
                call minikube docker-env --shell=cmd > docker_env.bat
                call docker_env.bat

                REM === Update deployment to use the new tag ===
                kubectl set image deployment/django-deployment django-container=${IMAGE_NAME}:${IMAGE_TAG} --record

                REM === Restart pods so new image actually runs ===
                kubectl rollout restart deployment/django-deployment
                kubectl rollout status deployment/django-deployment
                """
            }
        }
    }

    post {
        success {
            echo "✅ Deployment successful — running ${IMAGE_NAME}:${IMAGE_TAG}"
        }
        failure {
            echo "❌ Deployment failed — check console logs."
        }
    }
}
